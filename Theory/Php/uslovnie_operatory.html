Условные операторы

Оператор if else

Синтаксис

If (условие) {

// Код при выполнении условия

}

Else {

// Код при не выполнении условия

}



Пример

<?php

$a = 0;

$b = 3;

If ($a > $b) {

echo "a больше b";

}

Else {

echo "a меньше b";

}



Условный оператор switch

Синтаксис

Switch (имя переменной) {

case значение: код; break;

case значение: код; break;

default: код;

}



Пример

$a = 3;

Switch ( $a ) {

case 0: echo "a равно 0"; break;

case 1: echo "a равно 1"; break;

case 2: echo "a равно 2"; break;

default: echo "переменная не известна";

}

break – оператор остановки сравнения.
